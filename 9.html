css: #form-container { width: 80%; margin: 0 auto; } .step { display: none; } .step.active { display: block; } button { margin-top: 10px; } js: document.addEventListener('DOMContentLoaded', function() { const form = document.getElementById('multi-step-form'); const formSteps = form.querySelectorAll('.step'); let currentStep = 0; // Show the current step and update history function showStep(stepIndex) { formSteps.forEach((step, index) => { if (index === stepIndex) { step.classList.add('active'); } else { step.classList.remove('active'); } }); // Update history state history.pushState({ step: stepIndex }, `Step ${stepIndex + 1}`, `?step=${stepIndex + 1}`); } // Initialize form with the correct step based on URL query parameter function initForm() { const urlParams = new URLSearchParams(window.location.search); const stepParam = urlParams.get('step'); if (stepParam) { const stepIndex = parseInt(stepParam) - 1; if (!isNaN(stepIndex) && stepIndex >= 0 && stepIndex < formSteps.length) { currentStep = stepIndex; showStep(currentStep); } } } // Event listeners for next and previous buttons form.querySelectorAll('.next-step').forEach(button => { button.addEventListener('click', function(e) { e.preventDefault(); if (currentStep < formSteps.length - 1) { currentStep++; showStep(currentStep); } }); }); form.querySelectorAll('.prev-step').forEach(button => { button.addEventListener('click', function(e) { e.preventDefault(); if (currentStep > 0) { currentStep--; showStep(currentStep); } }); }); // Handle form submission (prevent default) form.addEventListener('submit', function(e) { e.preventDefault(); // Perform form submission logic here alert('Form submitted successfully!'); }); // Initialize the form initForm(); });
