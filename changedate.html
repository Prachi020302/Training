function daysBetweenDates(date1, date2) { // Convert both dates to milliseconds var date1Millis = date1.getTime(); var date2Millis = date2.getTime(); // Calculate the difference in milliseconds var differenceMillis = Math.abs(date2Millis - date1Millis); // Convert milliseconds to days (1 day = 24 hours * 60 minutes * 60 seconds * 1000 milliseconds) var daysDifference = Math.ceil(differenceMillis / (1000 * 60 * 60 * 24)); return daysDifference; } // Example usage: var specifiedDate = new Date('2024-01-01'); // Change this to the specified date var inputDate = new Date('2024-02-15'); // Change this to the input date var days = daysBetweenDates(inputDate, specifiedDate); console.log("Number of days between the specified date and the input date:", days); Days Between Dates
